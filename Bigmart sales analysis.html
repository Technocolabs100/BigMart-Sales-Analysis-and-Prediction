Dealing with the null values¶

Dealing with null values of column "Item_Weight"¶

Let us replace the null values from 'Item_Weight' to it's mean values:

In [14]:
IWM=train['Item_Weight'].mean()       #For train dataset.
IWm=test['Item_Weight'].mean()       #For test dataset.

print(IWM)
print(IWm)

12.695633368756374

test['Item_Weight']=test['Item_Weight'].fillna(value=IWm, inplace=False)

from sklearn.model_selection import cross_val_score
from sklearn.metrics import mean_squared_error

def train(model, X, y):
    model.fit(X,y)               # train the model
    pred=model.predict(X)        #predict the training set
    
    # Performing cross validation
    c_score=cross_val_score(model, X, y, scoring='neg_mean_squared_error')
    c_score=np.abs(np.mean(c_score))
    
    print('Model Report')
    print('MSE:', mean_squared_error(y,pred))
    print("Cross validation score:", c_score)
